http{
    server
	{
        server_name   server1;
        listen        8080;
        timeout 10;
        error_page    400 404  /40x.html;

        location	/images
		{
			alias	SERVER1/docs/images/;
			methods GET DELETE;
		}
		location        /docs
        {
            autoindex	on;
            methods 	GET;
            alias		SERVER1/docs/;
        }
		location	/post
		{
			methods		GET POST;
			alias		SERVER1/docs;
			postdir		SERVER1/uploads/;
		}
		location	/downloads
		{
			methods GET DELETE;
			alias	SERVER1/uploads/;
		}
        location    /cgi
        {
            cgi_pass    SERVER1/cgi/;
        }
        location    /redirect
        {
            redirect SERVER1/docs/;
        }
        location      /{
            alias      SERVER1/docs/;
            methods   GET;
            index   index.html index.htm;
        }
    }

	server

    }
